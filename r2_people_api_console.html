Quickstart · TypeScript
// Instale o SDK
npm install @r2-people/sdk

// Configure
import { R2People } from '@r2-people/sdk';
const client = new R2People({ apiKey: process.env.R2_API_KEY });

// Liste employees
const employees = await client.employees.list({
  status: 'active',
  limit: 50
});

// Crie movement com idempotência
const movement = await client.movements.create({
  employeeId: 'abc-...',
  type: 'PROMOTION',
  effectiveDate: '2026-06-01'
}, { idempotencyKey: crypto.randomUUID() });
